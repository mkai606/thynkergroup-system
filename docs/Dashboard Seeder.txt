const SystemSettingsContext = createContext();

export const useSystemSettings = () => useContext(SystemSettingsContext);

const SystemSettingsProvider = ({ children }) => {
  const [settings, setSettings] = useState(DEFAULT_SYSTEM_SETTINGS);

  const updateSetting = (section, field, value) => {
    setSettings(prev => ({
      ...prev,
      [section]: {
        ...prev[section],
        [field]: value
      }
    }));
  };

  const resetSettings = () => setSettings(DEFAULT_SYSTEM_SETTINGS);

  return (
    <SystemSettingsContext.Provider value={{ settings, updateSetting, resetSettings }}>
      {children}
    </SystemSettingsContext.Provider>
  );
};

// --- Utility Functions ---

const calculateTier = (followers, config) => {
  if (followers >= config.tierA_minFollowers) return 'A';
  if (followers >= config.tierB_minFollowers) return 'B';
  if (followers >= config.tierC_minFollowers) return 'C';
  if (followers >= config.tierD_minFollowers) return 'D';
  return 'E';
};

const calculateRating = (completed, submitted) => {
  if (!submitted || submitted === 0) return 0;
  return Math.min(5, ((completed / submitted) * 5).toFixed(1));
};

const formatCurrency = (amount) => `RM ${amount.toLocaleString()}`;

const addDays = (dateStr, days) => {
  const date = new Date(dateStr);
  date.setDate(date.getDate() + days);
  return date.toISOString().split('T')[0];
};

const RegistrationQueue = ({ registrations, onApprove, onReject, theme }) => {
  const [selected, setSelected] = useState(null);
  const [verifiedProfiles, setVerifiedProfiles] = useState({}); 
  const [verifyingId, setVerifyingId] = useState(null);
  const { settings } = useSystemSettings();

  useEffect(() => {
    if (registrations.length > 0 && !selected) setSelected(registrations[0]);
    if (registrations.length === 0) setSelected(null);
  }, [registrations, selected]);

  const isProfileVerified = (regId, index) => {
      if (registrations.find(r => r.id === regId)?.verificationStatus === 'Verified') return true;
      return verifiedProfiles[regId] && verifiedProfiles[regId].includes(index);
  };

  const handleVerifyProfile = (regId, index) => {
    setVerifyingId(index);
    setTimeout(() => {
        setVerifiedProfiles(prev => {
            const currentList = prev[regId] || [];
            if(currentList.includes(index)) return prev;
            return { ...prev, [regId]: [...currentList, index] };
        });
        setVerifyingId(null);
        if (selected && selected.id === regId) {
             selected.verificationStatus = 'Verified'; 
        }
    }, 1200);
  };

  if (registrations.length === 0) return (
    <div className="flex flex-col items-center justify-center h-full text-gray-500">
      <UserPlus size={64} className="mb-4 opacity-20" />
      <p className="text-xl font-medium font-anton uppercase">No Pending Registrations</p>
    </div>
  );

  const primaryProfile = selected ? selected.socialProfiles.reduce((prev, current) => (prev.followers > current.followers) ? prev : current) : null;
  const calculatedTier = primaryProfile ? calculateTier(primaryProfile.followers, settings.tierConfig) : 'E';
  const totalFollowers = selected ? selected.socialProfiles.reduce((acc, curr) => acc + curr.followers, 0) : 0;

  return (
    <div className="h-[calc(100vh-8rem)] flex gap-6">
      <div className={`w-1/3 ${theme.card} rounded-2xl flex flex-col overflow-hidden`}>
        <div className={`p-4 border-b ${theme.borderColor} ${theme.tableHeader} flex justify-between items-center`}>
          <h3 className={`font-semibold ${theme.textMain} uppercase tracking-wider`}>Pending Sidekicks ({registrations.length})</h3>
        </div>
        <div className="flex-1 overflow-auto divide-y divide-white/5">
          {registrations.map(reg => {
            const mainProfile = reg.socialProfiles.reduce((prev, current) => (prev.followers > current.followers) ? prev : current);
            return (
              <div 
                key={reg.id} 
                onClick={() => setSelected(reg)}
                className={`p-4 cursor-pointer transition-colors border-l-4 ${selected?.id === reg.id ? 'bg-[#CCFF00]/10 border-[#CCFF00]' : `${theme.tableRowHover} border-transparent`}`}
              >
                <div className="flex justify-between items-center mb-1">
                  <span className={`font-bold ${theme.textMain}`}>{reg.name}</span>
                  <span className="text-xs text-gray-500 font-mono">{reg.submittedAt}</span>
                </div>
                <p className={`text-sm ${theme.textMuted} mb-2 truncate`}>{mainProfile.handle}</p>
                <div className="flex items-center gap-2 flex-wrap">
                   <Badge color="blue">{mainProfile.platform}</Badge>
                   {reg.socialProfiles.length > 1 && (
                     <span className="text-[10px] bg-[#333] px-2 py-0.5 rounded text-gray-400">+{reg.socialProfiles.length - 1} more</span>
                   )}
                </div>
              </div>
            );
          })}
        </div>
      </div>

      <div className={`flex-1 ${theme.card} rounded-2xl p-8 flex flex-col overflow-y-auto`}>
        {selected ? (
          <>
            <div className="flex justify-between items-start mb-8">
              <div>
                <h2 className={`text-4xl ${theme.textMain} mb-1 tracking-wide`}>{selected.name}</h2>
                <div className="flex flex-col gap-1.5 mt-2">
                  <p className={`${theme.textMuted} uppercase tracking-wider text-xs font-bold`}>Applying for <span className={`${theme.accentLime}`}>Sidekick Account</span></p>
                  <div className="flex items-center text-sm text-gray-400 font-mono">

const TaskCreator = ({ onCreate, theme, sidekicks }) => {
    const { settings } = useSystemSettings();
    const [form, setForm] = useState({
        title: '',
        scope: '',
        platform: 'Instagram',
        tier: 'C',
        count: 50,
        payment: settings.campaignConfig.minPayout,
        deadline: '',
        instructions: '',
        hashtags: '',
        targetLink: '',
        directives: '',
        tngLink: '',
        tngQrImage: null,
        expReward: settings.campaignConfig.defaultExpReward, 
        vipOnly: false,
        minFollowersOverride: ''
    });

    const totalLiability = form.count * form.payment;
    
    // Dynamic calculation of eligible sidekicks
    const eligibleSidekicksCount = useMemo(() => {
        return sidekicks.filter(sk => {
            if (sk.platform !== form.platform) return false;
            if (form.vipOnly && sk.vipStatus !== 'Active') return false;

            let minFollowers = 0;
            switch(form.tier) {
                case 'A': minFollowers = settings.tierConfig.tierA_minFollowers; break;
                case 'B': minFollowers = settings.tierConfig.tierB_minFollowers; break;
                case 'C': minFollowers = settings.tierConfig.tierC_minFollowers; break;
                case 'D': minFollowers = settings.tierConfig.tierD_minFollowers; break;
                case 'E': minFollowers = settings.tierConfig.tierE_minFollowers; break;
                default: minFollowers = 0;
            }

            if (form.minFollowersOverride && !isNaN(form.minFollowersOverride) && form.minFollowersOverride !== '') {
                minFollowers = parseInt(form.minFollowersOverride, 10);
            }

            return sk.followers >= minFollowers;
        }).length;
    }, [sidekicks, form.platform, form.tier, form.vipOnly, form.minFollowersOverride, settings]);


    const handleSubmit = (e) => {
        e.preventDefault();
        onCreate(form);
    };

function SidekickCommandCenterInner() {
  const { settings } = useSystemSettings();
  const [currentView, setCurrentView] = useState('dashboard');
  const [selectedCampaign, setSelectedCampaign] = useState(null);
  const [sidekickSearch, setSidekickSearch] = useState('');
  const [globalSearch, setGlobalSearch] = useState(''); 
  const [tierFilter, setTierFilter] = useState('All');
  const [isDarkMode, setIsDarkMode] = useState(true);
  
  const theme = isDarkMode ? DARK_THEME : LIGHT_THEME;

  const [sidekicks, setSidekicks] = useState([]);
  const [tasks, setTasks] = useState(initialTasks);
  const [applications, setApplications] = useState(initialApplications);
  const [approvals, setApprovals] = useState(initialApprovals);
  const [registrations, setRegistrations] = useState(initialRegistrations);
  const [broadcasts, setBroadcasts] = useState(initialBroadcasts);
  const [notification, setNotification] = useState(null);

  useEffect(() => {
    // Generate sidekicks dynamically using context settings
    setSidekicks(generateSidekicks(50, settings));
  }, [settings.tierConfig, settings.vipConfig]);

  const showNotification = (message, type = 'success') => {
    setNotification({ message, type });
    setTimeout(() => setNotification(null), 3000);
  };

  const filteredTasks = tasks.filter(t => 
    t.title.toLowerCase().includes(globalSearch.toLowerCase()) ||
    t.platform.toLowerCase().includes(globalSearch.toLowerCase())
  );

  const filteredRegistrations = registrations.filter(r => 
    r.name.toLowerCase().includes(globalSearch.toLowerCase()) ||
    r.socialProfiles.some(p => p.handle.toLowerCase().includes(globalSearch.toLowerCase()))
  );

  const filteredApprovals = approvals.filter(a => 
    a.sidekickName.toLowerCase().includes(globalSearch.toLowerCase()) ||
    a.task.toLowerCase().includes(globalSearch.toLowerCase())
  );

  const filteredSidekicks = sidekicks.filter(sk => {
    const searchTerm = sidekickSearch || globalSearch; 
    const matchesSearch = sk.name.toLowerCase().includes(searchTerm.toLowerCase());
    const matchesTier = tierFilter === 'All' || sk.tier === tierFilter;
    return matchesSearch && matchesTier;
  });

  const vipRequests = useMemo(() => {
      return sidekicks.filter(sk => ['PaymentSubmitted', 'PendingApproval'].includes(sk.vipStatus));
  }, [sidekicks]);

  const handleCreateTask = (newTask) => {
    const budget = newTask.count * newTask.payment;
    const task = {
      id: tasks.length + 1,
      ...newTask,
      budget: budget,
      spent: 0,
      applied: 0,
      accepted: 0,
      completed: 0,
      status: 'Active',
      hiddenDetails: newTask.instructions,
      instructionsLocked: true
    };
    setTasks([task, ...tasks]);
    showNotification(`Campaign "${newTask.title}" broadcasted. Liability: ${formatCurrency(budget)}`);
    setCurrentView('tasks');
  };

  const handleApproveVip = (id) => {
      setSidekicks(prev => prev.map(sk => {
          if(sk.id === id) {
              const now = new Date().toISOString().split('T')[0];
              return {
                  ...sk,
                  level: 'VIP',
                  vipStatus: 'Active',
                  vipApprovedAt: now,
                  vipExpiresAt: addDays(now, settings.vipConfig.vipDurationDays),
                  verifiedBadge: true
              }
          }
          return sk;
      }));
      showNotification('VIP Request Approved');
  };

  const handleRejectVip = (id) => {
      setSidekicks(prev => prev.map(sk => sk.id === id ? {...sk, vipStatus: 'Eligible'} : sk));
      showNotification('VIP Request Rejected', 'error');
  };

  const handleApproveRegistration = (reg) => {
    const primaryProfile = reg.socialProfiles.reduce((prev, current) => (prev.followers > current.followers) ? prev : current);
    const tier = calculateTier(primaryProfile.followers, settings.tierConfig);
    
    const newSidekick = {
      id: `SK-${1000 + sidekicks.length + 1}`,
      name: reg.name,
      handle: primaryProfile.handle,
      tier: tier,
      followers: primaryProfile.followers,
      platform: primaryProfile.platform,
      rating: 0,
      completedTasks: 0,
      submittedTasks: 0,
      status: 'Active',
      earnings: 0,
      flagged: false,
      joinDate: new Date().toISOString().split('T')[0],
      level: 'Premium',
      vipStatus: 'NotEligible',
      verifiedBadge: false,
      totalExp: 0,
      monthlyExp: 0,
      approvedPaidTasks: 0,
      attemptedTasks: 0,
      successRate: 0,
      referralCode: `${reg.name.substring(0,3).toUpperCase()}${Date.now().toString().slice(-4)}`,
      referralCount: 0,
      campaignCompletedCount: 0
    };
    setSidekicks([newSidekick, ...sidekicks]);
    setRegistrations(prev => prev.filter(r => r.id !== reg.id));
    showNotification(`Sidekick ${reg.name} approved as Tier ${tier}.`);
  };

  const handleRejectRegistration = (id) => {
    setRegistrations(prev => prev.filter(r => r.id !== id));
    showNotification('Registration rejected.', 'error');
  };

  const handleApproveTask = (approvalId, taskId, expAwarded = 50, sidekickId) => {
    setApprovals(prev => prev.filter(item => item.id !== approvalId));
    setSidekicks(prev => prev.map(sk => {
        if(sk.id === sidekickId || sk.name === approvals.find(a => a.id === approvalId)?.sidekickName) { 
             const newApproved = (sk.approvedPaidTasks || 0) + 1;
             const newAttempted = (sk.attemptedTasks || 0); 
             return {
                 ...sk,
                 monthlyExp: (sk.monthlyExp || 0) + expAwarded,
                 totalExp: (sk.totalExp || 0) + expAwarded,
                 approvedPaidTasks: newApproved,
                 successRate: newAttempted > 0 ? (newApproved / newAttempted) * 100 : 0
             }
        }
        return sk;
    }));
    showNotification(`Payment released & ${expAwarded} EXP awarded!`, 'success');
  };

  const handleRejectTask = (id) => {
    setApprovals(prev => prev.filter(item => item.id !== id));
    showNotification('Submission rejected.', 'error');
  };

  const handleCampaignClick = (campaign) => {
    setSelectedCampaign(campaign);
    setCurrentView('campaign-detail');
  };

  const handleUpdateAppStatus = (appId, newStatus) => {
    setApplications(prev => prev.map(a => a.id === appId ? { ...a, status: newStatus } : a));
    if (selectedCampaign) {
        const adjustment = newStatus === 'Accepted' ? 1 : 0;
        setTasks(prev => prev.map(t => t.id === selectedCampaign.id ? { ...t, accepted: t.accepted + adjustment } : t));
        setSelectedCampaign(prev => ({ ...prev, accepted: prev.accepted + adjustment }));
    }
    showNotification(`Application marked as ${newStatus}`);
  };

  const handleSendBroadcast = (data) => {
      const typeMap = {
          'Admin HQ': 'HQ',
          'System Admin': 'Admin',
          'Sidekick Mentor': 'Mentor',
          'Community Mgr': 'Community'
      };

      const newBroadcast = {
          id: broadcasts.length + 1,
          senderType: typeMap[data.senderType] || 'Admin',
          sender: data.senderType.toUpperCase(),
          message: data.message,
          time: data.scheduleType === 'Now' ? 'Just now' : `Scheduled: ${data.scheduledDate}`,
          audience: data.audience,
          status: data.scheduleType === 'Now' ? 'Sent' : 'Scheduled'
      };

      setBroadcasts([newBroadcast, ...broadcasts]);
      showNotification(data.scheduleType === 'Now' ? 'Transmission Broadcasted' : 'Transmission Scheduled', 'success');
  };

  const toggleTheme = () => setIsDarkMode(!isDarkMode);

  const NavItem = ({ id, icon: Icon, label, count }) => (
    <button
      onClick={() => { setCurrentView(id); setSelectedCampaign(null); }}
      className={`w-full flex items-center px-4 py-3 text-xs font-bold uppercase tracking-widest rounded-r-full transition-all mb-2 border-l-4 ${

    >

      <span className="flex-1 text-left">{label}</span>

          {count}
        </span>
      )}
    </button>
  );

  return (
    <div className={`flex h-screen ${theme.bg} font-sans text-gray-100 overflow-hidden`}>
      <GlobalStyles isDarkMode={isDarkMode} />
      
      {/* Sidebar */}
      <aside className={`w-64 ${theme.sidebarBg} border-r ${theme.borderColor} flex flex-col flex-shrink-0 z-20`}>
        <div className="p-6 flex items-center space-x-3 mb-2">
          <div className="w-10 h-10 bg-[#CCFF00] rounded-sm flex items-center justify-center shadow-[0_0_15px_rgba(204,255,0,0.3)] flex-shrink-0">
            <Megaphone className="text-black" size={20} strokeWidth={3} />
          </div>
          <div className="flex flex-col">
            <span className={`text-2xl font-anton ${theme.textMain} tracking-wide leading-none`}>SIDEKICK</span>
            <span className="text-[10px] font-bold text-[#CCFF00] uppercase tracking-widest leading-none mt-1">Admin Command</span>
          </div>
        </div>

        <nav className="flex-1 px-0 py-6 space-y-1">
          <div className="text-[10px] font-bold text-gray-500 uppercase tracking-widest mb-4 px-6">Mission Control</div>
          <NavItem id="dashboard" icon={LayoutDashboard} label="Overview" />
          <NavItem id="tasks" icon={Megaphone} label="Campaigns" count={tasks.filter(t => t.status === 'Active').length} />
          <NavItem id="approvals" icon={CheckSquare} label="Approvals" count={approvals.length} />
          <NavItem id="registrations" icon={UserPlus} label="Registrations" count={registrations.length} />
          
          <div className="text-[10px] font-bold text-gray-500 uppercase tracking-widest mb-4 mt-8 px-6">Network Assets</div>
          <NavItem id="seeders" icon={Users} label="Sidekick Hub" />
          <NavItem id="vip-requests" icon={Crown} label="VIP Requests" count={vipRequests.length} />
          <NavItem id="ladderboard" icon={Trophy} label="Leaderboard" />
          <NavItem id="broadcasts" icon={Radio} label="Broadcasts" />
          <NavItem id="analytics" icon={BarChart3} label="Intel" />
        </nav>

        <div className={`p-4 border-t ${theme.borderColor}`}>
          <div className={`flex items-center p-3 rounded-xl ${isDarkMode ? 'bg-[#111]' : 'bg-gray-100'} border ${isDarkMode ? 'border-white/5' : 'border-gray-200'}`}>
            <div className={`w-8 h-8 rounded ${isDarkMode ? 'bg-[#333]' : 'bg-gray-300'} flex items-center justify-center border ${theme.borderColor}`}>
                <span className="font-bold text-[#CCFF00]">AD</span>
            </div>
            <div className="ml-3">
              <p className={`text-xs font-bold ${theme.textMain} uppercase tracking-wider`}>Admin User</p>
              <p className="text-[10px] text-gray-500 uppercase">Level 5 Access</p>
            </div>
          </div>
        </div>
      </aside>

      {/* Main Content */}
      <div className={`flex-1 flex flex-col min-w-0 overflow-hidden ${isDarkMode ? "bg-[url('https://www.transparenttextures.com/patterns/dark-matter.png')]" : "bg-gray-50"}`}>
        <header className={`${theme.headerBg} backdrop-blur-md border-b ${theme.borderColor} h-20 flex items-center justify-between px-8 z-10`}>
          <h1 className={`text-3xl font-anton ${theme.textMain} tracking-wide uppercase`}>
            {currentView === 'tasks' ? 'Campaign Ops' : 
             currentView === 'campaign-detail' ? 'Mission Brief' :
             currentView === 'registrations' ? 'Recruitment' :
             currentView === 'seeders' ? 'Sidekick Hub' : 
             currentView === 'approvals' ? 'Verification' : 
             currentView === 'create-task' ? 'Initiate Mission' :
             currentView === 'analytics' ? 'Intelligence' :
             currentView === 'broadcasts' ? 'HQ Broadcasts' :
             currentView === 'vip-requests' ? 'VIP Approval' :
             currentView === 'ladderboard' ? 'Leaderboard' :
             'Command Center'}
          </h1>
          
          <div className="flex items-center space-x-6">
            <div className="relative hidden md:block group">
              <input 
                type="text" 
                placeholder="SEARCH DATABASE..." 
                value={globalSearch}
                onChange={(e) => setGlobalSearch(e.target.value)}
                className={`w-64 pl-10 pr-4 py-2 rounded-full ${theme.input} text-xs placeholder-gray-500 uppercase tracking-wider transition-all`}
              />
              <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-500 group-focus-within:text-[#CCFF00]" size={14} />
            </div>
            
            <button onClick={toggleTheme} className="p-2 text-gray-400 hover:text-[#CCFF00] transition-colors" title="Toggle Theme">
               {isDarkMode ? <Sun size={20}/> : <Moon size={20}/>}
            </button>

            <button className="p-2 text-gray-400 hover:text-[#CCFF00] relative transition-colors">
              <Bell size={20} />
              <span className="absolute top-1.5 right-1.5 w-2 h-2 bg-[#CCFF00] rounded-full animate-pulse"></span>
            </button>
          </div>
        </header>

        <main className="flex-1 overflow-auto p-8 relative">
          {currentView === 'dashboard' && (
             <div className="space-y-8">
                <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
                  <div className={`${theme.card} p-6 rounded-2xl relative overflow-hidden group`}>
                     <div className="absolute right-0 top-0 p-3 opacity-10 group-hover:opacity-20 transition-opacity">
                         <Users size={80} className={isDarkMode ? "text-white" : "text-gray-300"}/>
                     </div>
                     <div>
                        <p className={`text-xs ${theme.textMuted} uppercase tracking-widest`}>Active Sidekicks</p>
                        <h3 className={`text-4xl ${theme.textMain} font-anton mt-2`}>{sidekicks.length}</h3>
                     </div>
                     <div className="mt-4 w-12 h-1 bg-[#CCFF00]"></div>
                  </div>
                  <div className={`${theme.card} p-6 rounded-2xl relative overflow-hidden group`}>
                     <div className="absolute right-0 top-0 p-3 opacity-10 group-hover:opacity-20 transition-opacity">
                         <UserPlus size={80} className={isDarkMode ? "text-white" : "text-gray-300"}/>
                     </div>
                     <div>
                        <p className={`text-xs ${theme.textMuted} uppercase tracking-widest`}>Recruitment</p>
                        <h3 className={`text-4xl ${theme.textMain} font-anton mt-2`}>{registrations.length}</h3>
                     </div>
                     <div className="mt-4 w-12 h-1 bg-amber-500"></div>
                  </div>
                   <div className={`${theme.card} p-6 rounded-2xl relative overflow-hidden group`}>
                     <div className="absolute right-0 top-0 p-3 opacity-10 group-hover:opacity-20 transition-opacity">
                         <CheckSquare size={80} className={isDarkMode ? "text-white" : "text-gray-300"}/>
                     </div>
                     <div>
                        <p className={`text-xs ${theme.textMuted} uppercase tracking-widest`}>Pending Review</p>
                        <h3 className={`text-4xl ${theme.textMain} font-anton mt-2`}>{approvals.length}</h3>
                     </div>
                     <div className="mt-4 w-12 h-1 bg-emerald-500"></div>
                  </div>
                   <div className={`${theme.cardSecondary} p-6 rounded-2xl relative overflow-hidden group`}>
                     <div className="absolute right-0 top-0 p-3 opacity-10 group-hover:opacity-20 transition-opacity">
                         <Wallet size={80} className="text-[#CCFF00]"/>
                     </div>
                     <div>
                        <p className={`text-xs ${isDarkMode ? 'text-[#CCFF00]/80' : 'text-[#4D7C0F]/80'} uppercase tracking-widest`}>Monthly Liability</p>
                        <h3 className={`text-4xl ${isDarkMode ? 'text-[#CCFF00]' : 'text-[#4D7C0F]'} font-anton mt-2`}>RM 12.4K</h3>
                     </div>
                     <div className={`mt-4 w-12 h-1 ${isDarkMode ? 'bg-[#CCFF00]' : 'bg-[#4D7C0F]'}`}></div>
                  </div>
                </div>

                <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
                   <div className={`lg:col-span-2 ${theme.card} p-6 rounded-2xl`}>
                      <div className="flex justify-between items-center mb-6">
                          <h3 className={`font-bold ${theme.textMain} uppercase tracking-wider`}>Engagement Velocity</h3>
                          <Badge color="green">+12% vs LW</Badge>
                      </div>
                      <div className="h-64">
                         <ResponsiveContainer width="100%" height="100%">
                           <LineChart data={analyticsData}>
                             <CartesianGrid strokeDasharray="3 3" vertical={false} stroke={isDarkMode ? "#333" : "#e5e7eb"} />
                             <XAxis dataKey="name" axisLine={false} tickLine={false} tick={{fill: isDarkMode ? '#666' : '#9ca3af', fontSize: 10}} />
                             <YAxis axisLine={false} tickLine={false} tick={{fill: isDarkMode ? '#666' : '#9ca3af', fontSize: 10}} />
                             <Tooltip contentStyle={{backgroundColor: isDarkMode ? '#111' : '#fff', border: `1px solid ${isDarkMode ? '#333' : '#e5e7eb'}`, color: isDarkMode ? '#fff' : '#000'}} itemStyle={{color: '#CCFF00'}} />
                             <Line type="monotone" dataKey="reach" stroke="#CCFF00" strokeWidth={3} dot={{r:4, fill: isDarkMode ? '#111' : '#fff', strokeWidth: 2}} activeDot={{r:6, fill: '#CCFF00'}} />
                           </LineChart>
                         </ResponsiveContainer>
                      </div>
                   </div>
                   <div className={`${theme.card} p-6 rounded-2xl`}>
                      <h3 className={`font-bold ${theme.textMain} mb-4 flex items-center uppercase tracking-wider`}>
                          <ShieldAlert size={18} className="mr-2 text-red-500"/> Threat Monitor

                      <div className="space-y-3">
                        {systemAlerts.map(alert => (
                           <div key={alert.id} className={`p-4 rounded border text-xs leading-relaxed ${alert.type === 'error' ? 'bg-red-900/10 border-red-900/30 text-red-400' : 'bg-yellow-900/10 border-yellow-900/30 text-yellow-600'}`}>
                              <span className="font-bold block mb-1 uppercase tracking-wider">{alert.type} DETECTED</span>
                              {alert.message}
                              <div className="text-[10px] opacity-60 mt-2 font-mono">{alert.time}</div>
                           </div>
                        ))}
                      </div>
                   </div>
                </div>
             </div>
          )}

          {currentView === 'tasks' && (
            <div className="space-y-8">
               <div className="grid grid-cols-3 gap-6">
                 <div className={`${theme.cardSecondary} p-6 rounded-2xl relative overflow-hidden`}>
                    <p className={`opacity-80 text-xs font-bold uppercase tracking-widest mb-2 ${isDarkMode ? 'text-[#CCFF00]' : 'text-[#4D7C0F]'}`}>Total Liability</p>
                    <h3 className={`text-5xl font-anton ${isDarkMode ? 'text-[#CCFF00]' : 'text-[#4D7C0F]'}`}>{formatCurrency(tasks.reduce((acc, t) => acc + t.budget, 0))}</h3>
                 </div>
                 <div className={`${theme.card} p-6 rounded-2xl`}>
                    <p className={`text-gray-500 text-xs font-bold uppercase tracking-widest mb-2`}>Allocated Budget</p>
                    <h3 className={`text-5xl font-anton ${theme.textMain}`}>{formatCurrency(tasks.reduce((acc, t) => acc + t.spent, 0))}</h3>
                 </div>
                 <button onClick={() => setCurrentView('create-task')} className={`rounded-2xl border border-dashed ${isDarkMode ? 'border-gray-700' : 'border-gray-300'} hover:border-[#CCFF00] hover:bg-[#CCFF00]/5 flex flex-col items-center justify-center transition-all group`}>
                    <Plus className="text-gray-500 group-hover:text-[#CCFF00] mb-2 transition-colors" size={32} />
                    <span className="font-bold text-gray-500 group-hover:text-[#CCFF00] uppercase tracking-wider text-xs transition-colors">Create Mission</span>
                 </button>
               </div>

               <div className={`${theme.card} rounded-2xl overflow-hidden`}>
                  <div className={`p-4 border-b ${theme.borderColor} flex justify-between items-center ${theme.tableHeader}`}>
                    <h3 className={`font-semibold ${theme.textMain} uppercase tracking-wider`}>Active Operations</h3>
                  </div>
                  <table className="w-full text-left text-sm">
                    <thead className={`${theme.tableHeader} border-b ${theme.borderColor} uppercase text-xs tracking-wider`}>
                      <tr>
                        <th className="px-6 py-4 font-normal">Mission Name</th>
                        <th className="px-6 py-4 font-normal">Platform</th>
                        <th className="px-6 py-4 font-normal">Reward</th>
                        <th className="px-6 py-4 font-normal">Cost</th>
                        <th className="px-6 py-4 font-normal">Status</th>
                        <th className="px-6 py-4 font-normal text-right">Command</th>
                      </tr>
                    </thead>
                    <tbody className={`divide-y ${isDarkMode ? 'divide-white/5' : 'divide-gray-200'}`}>
                      {filteredTasks.length > 0 ? (
                          filteredTasks.map(task => (
                            <tr key={task.id} className={`${theme.tableRowHover} cursor-pointer transition-colors`} onClick={() => handleCampaignClick(task)}>
                              <td className={`px-6 py-4 font-bold ${theme.textMain}`}>
                                  {task.title}
                                  {task.vipOnly && <span className="ml-2 text-[10px] bg-yellow-500/20 text-yellow-500 px-1.5 py-0.5 rounded border border-yellow-500/50">VIP</span>}
                              </td>
                              <td className="px-6 py-4"><Badge color="blue">{task.platform}</Badge></td>
                              <td className="px-6 py-4">
                                  <div className="flex items-center text-xs font-mono text-[#CCFF00]">
                                      <Award size={12} className="mr-1"/> {task.expReward} EXP
                                  </div>
                              </td>
                              <td className="px-6 py-4">
                                 <span className={`block ${theme.textMain} font-mono font-bold`}>{formatCurrency(task.spent)}</span>
                                 <span className="text-[10px] text-gray-500 uppercase">MAX {formatCurrency(task.budget)}</span>
                              </td>
                              <td className="px-6 py-4"><Badge color={task.status === 'Active' ? 'green' : 'gray'}>{task.status}</Badge></td>
                              <td className="px-6 py-4 text-right">
                                 <button className={`${theme.accentBlue} font-bold text-xs uppercase hover:text-current transition-colors`}>Execute</button>
                              </td>
                            </tr>
                          ))
                      ) : (
                          <tr><td colSpan="6" className="px-6 py-8 text-center text-gray-500 uppercase tracking-widest text-xs">No active operations found</td></tr>
                      )}
                    </tbody>
                  </table>
               </div>
            </div>
          )}

          {currentView === 'campaign-detail' && <CampaignDetailView campaign={selectedCampaign} applications={applications} onBack={() => setCurrentView('tasks')} onUpdateAppStatus={handleUpdateAppStatus} theme={theme} />}

          {currentView === 'registrations' && <RegistrationQueue registrations={filteredRegistrations} onApprove={handleApproveRegistration} onReject={handleRejectRegistration} theme={theme} />}
          
          {currentView === 'approvals' && <ApprovalQueue items={filteredApprovals} onApprove={handleApproveTask} onReject={handleRejectTask} theme={theme} />}

          {currentView === 'seeders' && (
             <div className={`${theme.card} rounded-2xl overflow-hidden`}>
                <div className={`p-4 border-b ${theme.borderColor} ${theme.tableHeader} flex justify-between items-center`}>
                    <h3 className={`font-semibold ${theme.textMain} uppercase tracking-wider`}>Sidekick Hub Roster ({filteredSidekicks.length})</h3>
                    <div className="flex items-center gap-4">
                        <div className="relative">
                            <Filter className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-500 pointer-events-none" size={14} />
                            <select 
                                value={tierFilter}
                                onChange={(e) => setTierFilter(e.target.value)}
                                className={`pl-10 pr-8 py-2 rounded-lg ${theme.input} text-xs uppercase tracking-wider appearance-none cursor-pointer hover:bg-opacity-80 transition-all`}
                            >
                                <option value="All">All Tiers</option>
                                <option value="A">Tier A</option>
                                <option value="B">Tier B</option>
                                <option value="C">Tier C</option>
                                <option value="D">Tier D</option>
                                <option value="E">Tier E</option>
                            </select>
                            <div className="absolute right-3 top-1/2 transform -translate-y-1/2 pointer-events-none">
                                <svg width="10" height="6" viewBox="0 0 10 6" fill="none" xmlns="http://www.w3.org/2000/svg">
                                    <path d="M1 1L5 5L9 1" stroke="#666" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round"/>
                                </svg>
                            </div>
                        </div>

                        <button className={`${theme.btnSecondary} px-4 py-2 rounded text-xs`}>Export Intel</button>
                    </div>
                </div>
                <table className="w-full text-left text-sm">
                   <thead className={`${theme.tableHeader} border-b ${theme.borderColor} uppercase text-xs tracking-wider`}>
                      <tr>
                         <th className="px-6 py-4 font-normal">SIDEKICK NAME</th>
                         <th className="px-6 py-4 font-normal">TIER</th>
                         <th className="px-6 py-4 font-normal">Level</th>
                         <th className="px-6 py-4 font-normal">Success Rate</th>
                         <th className="px-6 py-4 font-normal text-right">Total EXP</th>
                      </tr>
                   </thead>
                   <tbody className={`divide-y ${isDarkMode ? 'divide-white/5' : 'divide-gray-200'}`}>
                      {filteredSidekicks.length > 0 ? (
                          filteredSidekicks.map(sk => (
                             <tr key={sk.id} className={`${theme.tableRowHover} transition-colors`}>
                                <td className="px-6 py-4">
                                   <div className="flex items-center">
                                        <div>
                                             <div className={`font-bold ${theme.textMain} flex items-center gap-1`}>
                                                {sk.name}
                                                {sk.verifiedBadge && <CheckCircle size={14} className="text-[#CCFF00] fill-current"/>}
                                             </div>
                                             <div className="text-xs text-gray-500 font-mono">{sk.handle}</div>
                                        </div>
                                        {sk.flagged && (
                                            <div className="ml-2 text-red-500" title="Flagged for suspicious activity">
                                                 <AlertTriangle size={16}/>
                                            </div>
                                        )}
                                   </div>
                                </td>
                                <td className="px-6 py-4"><Badge>{sk.tier}</Badge></td>
                                <td className="px-6 py-4">
                                    <Badge color={sk.level === 'VIP' ? 'gold' : 'gray'}>{sk.level}</Badge>
                                </td>
                                <td className="px-6 py-4 text-gray-500 font-mono">
                                   {(sk.successRate || 0).toFixed(1)}%
                                </td>
                                <td className="px-6 py-4 text-right text-[#CCFF00] font-bold font-mono">{sk.totalExp.toLocaleString()}</td>
                             </tr>
                          ))
                      ) : (
                          <tr>
                              <td colSpan="5" className="px-6 py-8 text-center text-gray-500 uppercase tracking-widest text-xs">
                                  No sidekicks found matching search.
                              </td>
                          </tr>
                      )}
                   </tbody>
                </table>
             </div>
          )}
          
          {currentView === 'vip-requests' && <VipRequestQueue requests={vipRequests} onApprove={handleApproveVip} onReject={handleRejectVip} theme={theme} />}

          {currentView === 'ladderboard' && <Leaderboard sidekicks={sidekicks} theme={theme} />}

          {currentView === 'broadcasts' && <BroadcastCenter broadcasts={broadcasts} onSend={handleSendBroadcast} theme={theme} isDarkMode={isDarkMode} />}

          {currentView === 'create-task' && <TaskCreator onCreate={handleCreateTask} theme={theme} sidekicks={sidekicks} />}
          
          {currentView === 'analytics' && <IntelCenter theme={theme} analyticsData={analyticsData} systemAlerts={systemAlerts} isDarkMode={isDarkMode} onShowNotification={showNotification} />}

        </main>
      </div>

      {notification && (
        <div className={`fixed bottom-6 right-6 px-6 py-4 rounded-xl shadow-2xl text-white transform transition-all duration-300 translate-y-0 z-50 border ${
          notification.type === 'success' ? 'bg-[#111] border-[#CCFF00] text-[#CCFF00]' : 'bg-[#111] border-red-500 text-red-500'
        }`}>
          <div className="flex items-center">
            {notification.type === 'success' ? <CheckCircle size={20} className="mr-3" /> : <XCircle size={20} className="mr-3" />}
            <span className="font-bold uppercase tracking-wider text-sm">{notification.message}</span>
          </div>
        </div>
      )}
    </div>
  );
}
