const Badge = ({ children, color = 'blue', className = '', icon: Icon }) => {
  const colors = {
    blue: 'bg-blue-900/40 text-blue-300 border-blue-700',
    green: 'bg-[#3D4421] text-[#CCFF00] border-[#CCFF00]', 
    neon: 'bg-[#CCFF00] text-black border-[#CCFF00] shadow-[0_0_10px_rgba(204,255,0,0.3)]',
    yellow: 'bg-yellow-900/40 text-yellow-300 border-yellow-700', 
    purple: 'bg-purple-900/40 text-purple-300 border-purple-700',
    gray: 'bg-gray-800 text-gray-400 border-gray-700',
    red: 'bg-red-900/40 text-red-300 border-red-700',
  };
  return (
    <span className={`px-2 py-0.5 rounded flex items-center gap-1 text-[10px] uppercase font-bold tracking-wider border ${colors[color]} ${className} font-body`}>
      {Icon && <Icon size={10} />}
      {children}
    </span>
  );
};

  switch (platform) {
    case 'Instagram': return <Instagram size={16} className="text-pink-400" />;
    case 'Facebook': return <Facebook size={16} className="text-blue-400" />;
    case 'TikTok': return <div className="w-4 h-4 bg-white text-black rounded-full flex items-center justify-center text-[10px] font-bold">T</div>;
    case 'Youtube': return <Youtube size={16} className="text-red-500" />;
    default: return <Briefcase size={16} className="text-gray-400" />;

};

const StatusBadge = ({ status }) => {
  switch (status) {
    case 'applied': return <Badge color="blue">Applied</Badge>;
    case 'accepted': return <Badge color="purple">In Progress</Badge>;
    case 'submitted': return <Badge color="yellow">Under Review</Badge>;
    case 'approved': return <Badge color="green">Action Required</Badge>;
    case 'paid': return <Badge color="gray">Settled</Badge>;
    case 'rejected': return <Badge color="red">Rejected</Badge>;
    default: return <Badge color="gray">{status}</Badge>;
  }
};

      <div className="flex flex-col items-center mb-10 pt-4 relative">
        <div className="relative mb-5">
            <div className="w-28 h-28 rounded-full bg-[#262626] overflow-hidden border-2 border-[#CCFF00] p-1 shadow-[0_0_20px_rgba(204,255,0,0.15)]">
              <img src={user.avatar} alt="Profile" className="w-full h-full object-cover rounded-full" />
            </div>
            {user.sidekick_level === 'vip' && (
                <div className="absolute bottom-0 left-0 z-10 bg-[#121212] rounded-full p-1 border border-[#121212]">
                    <VerifiedBadge size={28} />
                </div>
            )}
        </div>
        
        <div className="flex items-center gap-2 mb-1">
          <h1 className="text-3xl font-display text-white uppercase tracking-wide">{user.name}</h1>
        </div>
        
        <p className="text-gray-500 text-xs font-mono mb-3">{user.handle}</p>
        <div className="flex gap-2">
          <Badge color={user.sidekick_level === 'vip' ? 'neon' : 'gray'}>
             SIDEKICK {user.sidekick_level}
          </Badge>
          <Badge color="purple">Rank #{user.rank_position}</Badge>
        </div>
      </div>

      <div className="grid grid-cols-2 gap-4 mb-8">
        <div className="bg-[#262626] p-4 rounded-xl border border-gray-700 text-center">
          <span className="block text-3xl font-display text-white flex items-center justify-center gap-1">
             <Trophy size={18} className="text-[#CCFF00]"/> {user.total_exp}
          </span>
          <span className="text-[10px] text-gray-500 font-bold uppercase tracking-widest">Total EXP</span>
        </div>
        <div className="bg-[#262626] p-4 rounded-xl border border-gray-700 text-center">
          <span className="block text-3xl font-display text-[#CCFF00]">{user.success_rate}</span>
          <span className="text-[10px] text-gray-500 font-bold uppercase tracking-widest">Success Ratio</span>
        </div>
      </div>

      {/* REFERRAL SYSTEM */}
      <div className="bg-[#1A1A1A] p-5 rounded-xl border border-dashed border-gray-700 mb-8">
         <div className="flex justify-between items-center mb-4">
             <h3 className="font-bold text-gray-400 uppercase text-xs tracking-wider">Referral Code</h3>
             <Badge color="green">{user.referral_count} Joined</Badge>
         </div>
         <div className="flex gap-2">
            <div className="flex-1 bg-black border border-gray-800 rounded-lg flex items-center justify-center p-3">
               <span className="font-mono text-xl text-white font-bold tracking-widest">{user.referral_code}</span>

            <button className="bg-[#262626] px-4 rounded-lg border border-gray-700 text-white hover:border-[#CCFF00]">
               <Copy size={20} />
            </button>
         </div>
      </div>

      {/* VIP UPGRADE CARD */}
      {user.sidekick_level === 'premium' && (
        <div className="bg-gradient-to-br from-[#121212] to-[#1A1A1A] rounded-xl border border-[#CCFF00] p-6 mb-8 relative overflow-hidden">
           <div className="absolute top-0 right-0 p-4 opacity-10"><Crown size={80} /></div>
           <h3 className="font-display text-2xl text-white uppercase italic mb-1">Become VIP</h3>
           <p className="text-xs text-gray-400 mb-6 max-w-[200px]">Unlock high-ticket campaigns, verified badge, and priority payments.</p>
           
           <div className="space-y-4 mb-6">
              <div>
                 <div className="flex justify-between text-[10px] uppercase font-bold text-gray-500 mb-1">
                    <span>Completed Campaigns</span>
                    <span className={user.completed_tasks >= 5 ? 'text-[#CCFF00]' : ''}>{user.completed_tasks}/5</span>
                 </div>
                 <div className="h-1.5 bg-gray-800 rounded-full overflow-hidden">
                    <div className="h-full bg-[#CCFF00]" style={{ width: `${Math.min(100, (user.completed_tasks/5)*100)}%` }}></div>
                 </div>
              </div>
              <div>
                 <div className="flex justify-between text-[10px] uppercase font-bold text-gray-500 mb-1">
                    <span>Referrals</span>
                    <span className={user.referral_count >= 5 ? 'text-[#CCFF00]' : ''}>{user.referral_count}/5</span>
                 </div>
                 <div className="h-1.5 bg-gray-800 rounded-full overflow-hidden">
                    <div className="h-full bg-[#CCFF00]" style={{ width: `${Math.min(100, (user.referral_count/5)*100)}%` }}></div>
                 </div>
              </div>
           </div>

           {user.vip_status === 'payment_submitted' ? (
              <button disabled className="w-full py-3 rounded-lg font-bold uppercase text-sm bg-gray-800 text-gray-500 border border-gray-700 cursor-wait">
                 Pending Admin Approval...
              </button>
           ) : (
             <button 
               onClick={() => setShowVipPaymentModal(true)}
               disabled={user.completed_tasks < 5 || user.referral_count < 5}
               className={`w-full py-3 rounded-lg font-bold uppercase text-sm flex items-center justify-center gap-2 transition-all
                  ${(user.completed_tasks >= 5 && user.referral_count >= 5) 
                    ? 'bg-[#CCFF00] text-black hover:bg-[#b3e600] shadow-[0_0_15px_rgba(204,255,0,0.3)]'

             >
                {(user.completed_tasks >= 5 && user.referral_count >= 5) ? 'Pay RM15 & Upgrade' : 'Requirements Not Met'}
             </button>
           )}
        </div>
      )}

      {/* Settings */}
      <div className="flex items-center space-x-2 mb-4">
        <div className="h-4 w-1 bg-gray-600"></div>
        <h3 className="font-display text-xl text-gray-400 uppercase tracking-wide">System</h3>
      </div>
      
      <button className="w-full py-4 text-red-400 text-xs font-bold uppercase tracking-widest bg-red-900/10 border border-red-900/30 rounded-full hover:bg-red-900/20 transition-colors">
        Deactivate Session
      </button>
    </div>

  const MyTasks = () => {
    const active = myApplications.filter(app => ['applied', 'accepted', 'submitted', 'under_review'].includes(app.status));
    const history = myApplications.filter(app => ['approved', 'paid', 'rejected'].includes(app.status));

    return (
      <div className="pb-20">
        <header className="flex justify-between items-center mb-6 pt-2">
            <h1 className="text-4xl font-display text-white uppercase tracking-wide">My Ops</h1>
        </header>
        
        {active.length === 0 && (
          <div className="text-center py-12 bg-[#262626] rounded-2xl border border-dashed border-gray-700">

            <button 
              onClick={() => setActiveTab('feed')}
              className="mt-6 text-[#CCFF00] font-bold text-xs uppercase tracking-widest border-b border-[#CCFF00] pb-1"
            >
              Initiate New Task
            </button>
          </div>
        )}

        <div className="space-y-3">
          {active.map(app => (
            <div 
              key={app.id}
              onClick={() => {
                const fullTask = INITIAL_TASKS.find(t => t.id === app.taskId) || {
                  title: app.taskTitle, reward: app.reward, exp_reward: app.exp || 0, brand: app.brand, platform: app.platform,
                  description: 'Detailed instructions loaded...', instructions: ['Post content', 'Tag brand'], hashtags: ['#ad']
                };
                setSelectedTask(fullTask);
              }}
              className="bg-[#262626] p-4 rounded-xl border border-gray-700 hover:border-[#CCFF00] transition-colors flex justify-between items-center cursor-pointer"
            >
              <div>
                <div className="flex items-center space-x-3 mb-2">
                  <h4 className="font-body font-bold text-white uppercase text-sm">{app.taskTitle}</h4>
                  <PlatformIcon platform={app.platform} />
                </div>
                <div className="flex items-center space-x-3">
                  <StatusBadge status={app.status} />
                  <span className="text-[10px] text-gray-500 uppercase font-bold tracking-wider">Due Soon</span>
                </div>
              </div>
              <div className="w-8 h-8 rounded-full bg-[#1A1A1A] flex items-center justify-center border border-gray-700">
                <ChevronRight className="text-gray-400" size={16} />
              </div>
            </div>
          ))}
        </div>

        {history.length > 0 && (
          <>
            <div className="flex items-center space-x-2 mb-2 mt-8">
                <div className="h-4 w-1 bg-gray-600"></div>
                <h3 className="font-display text-xl text-gray-400 uppercase tracking-wide">Mission Log</h3>
            </div>
            <div className="space-y-3">
              {history.map(app => (
                <div 
                  key={app.id}
                  onClick={() => {
                      const fullTask = INITIAL_TASKS.find(t => t.id === app.taskId) || {
                       title: app.taskTitle, reward: app.reward, exp_reward: app.exp || 0, brand: app.brand, platform: app.platform,
                       description: 'Historical task.', instructions: [], hashtags: []
                     };
                     setSelectedTask(fullTask);
                  }}
                  className="bg-[#1A1A1A] p-4 rounded-xl border border-gray-800 flex justify-between items-center opacity-80 hover:opacity-100"
                >
                  <div>
                    <h4 className="font-body font-bold text-gray-300 text-sm uppercase">{app.taskTitle}</h4>
                    <span className="text-[10px] text-gray-600 font-bold uppercase tracking-wider">{app.date}</span>
                  </div>
                  <div className="text-right">
                    <span className={`block font-display text-lg tracking-wide ${app.status === 'rejected' ? 'text-red-900 line-through' : 'text-[#3D4421] text-opacity-100 text-green-700'}`}>RM {app.reward}</span>
                    <span className="text-[9px] uppercase font-bold text-gray-600 tracking-widest">{app.status}</span>
                  </div>
                </div>
              ))}
            </div>
          </>
        )}
      </div>
    );
  };

  // --- RENDER ---

  return (
    <div className="min-h-screen bg-gray-900 font-sans text-gray-100 flex justify-center">
      {/* Font Injection */}
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Anton&family=Montserrat:wght@400;500;600;700;800&display=swap');
        
        .font-display { font-family: 'Anton', sans-serif; }
        .font-body { font-family: 'Montserrat', sans-serif; }
        
        .custom-scrollbar::-webkit-scrollbar { width: 4px; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #333; border-radius: 4px; }
        @keyframes slide-up {
          from { transform: translateY(100%); }
          to { transform: translateY(0); }
        }
        @keyframes slide-down {
          from { transform: translateY(-100%); opacity: 0; }
          to { transform: translateY(0); opacity: 1; }
        }
        .animate-slide-up { animation: slide-up 0.3s cubic-bezier(0.16, 1, 0.3, 1) forwards; }
        .animate-slide-down { animation: slide-down 0.3s cubic-bezier(0.16, 1, 0.3, 1) forwards; }
      `}</style>

      {/* Mobile Container */}
      <div className="w-full max-w-md bg-[#121212] min-h-screen shadow-2xl relative overflow-hidden border-x border-gray-800">
        
        {/* Top Notification Toast */}
        {notificationMsg && (
          <div className="absolute top-6 left-4 right-4 bg-[#CCFF00] text-black px-5 py-4 rounded-lg shadow-2xl z-50 flex items-center animate-slide-down border-2 border-white">
            <CheckCircle size={20} className="text-black mr-3" />
            <span className="text-xs font-bold uppercase tracking-wider">{notificationMsg}</span>
          </div>
        )}

        {/* Content Area */}
        <div className="h-full overflow-y-auto px-5 pt-6 custom-scrollbar">
          {activeTab === 'feed' && <TaskFeed />}
          {activeTab === 'tasks' && <MyTasks />}
          {activeTab === 'wallet' && <WalletScreen />}
          {activeTab === 'leaderboard' && <LeaderboardScreen />}
          {activeTab === 'profile' && <ProfileScreen />}
        </div>

        {/* Modals */}
        {selectedTask && (
          <TaskDetailView 
            task={selectedTask} 
            onClose={() => setSelectedTask(null)}
            onApply={handleApply}
            onSimulateApproval={handleSimulateApproval}
            onSubmitProof={handleSubmitProof}
            userApplications={myApplications}
            currentUser={user}
            onClaim={(app) => setClaimingApp(app)}
          />
        )}

        {showBroadcasts && <BroadcastCenter broadcasts={broadcasts} onClose={() => setShowBroadcasts(false)} />}
        {showQR && <QRModal onClose={() => setShowQR(false)} />}
        {activeBroadcast && <IncomingTransmissionModal message={activeBroadcast} onClose={() => setActiveBroadcast(null)} />}
        {claimingApp && <ClaimQRModal claimingApp={claimingApp} onClose={() => setClaimingApp(null)} onConfirm={handleConfirmClaim} />}
        {showVipPaymentModal && <VipPaymentModal onClose={() => setShowVipPaymentModal(false)} onConfirm={handleVipRequest} />}

        {/* Bottom Navigation */}
        <div className="absolute bottom-0 left-0 right-0 bg-[#1A1A1A] border-t border-gray-800 px-6 py-4 pb-6 flex justify-between items-center z-40">
           <button 
            onClick={() => setActiveTab('feed')}
            className={`flex flex-col items-center space-y-1 transition-colors ${activeTab === 'feed' ? 'text-[#CCFF00]' : 'text-gray-500 hover:text-gray-300'}`}
           >
             <Home size={24} strokeWidth={activeTab === 'feed' ? 2.5 : 2} />
             <span className="text-[9px] font-bold uppercase tracking-widest">Feed</span>
           </button>
           
           <button 
            onClick={() => setActiveTab('tasks')}
            className={`flex flex-col items-center space-y-1 transition-colors ${activeTab === 'tasks' ? 'text-[#CCFF00]' : 'text-gray-500 hover:text-gray-300'}`}
           >
             <div className="relative">
               <Briefcase size={24} strokeWidth={activeTab === 'tasks' ? 2.5 : 2} />
               {myApplications.filter(a => a.status === 'accepted').length > 0 && (
                 <span className="absolute -top-1 -right-1 w-2.5 h-2.5 bg-[#CCFF00] rounded-full border-2 border-[#1A1A1A]"></span>
               )}
             </div>
             <span className="text-[9px] font-bold uppercase tracking-widest">Ops</span>
           </button>

           <button 
            onClick={() => setActiveTab('wallet')}
            className={`flex flex-col items-center space-y-1 transition-colors ${activeTab === 'wallet' ? 'text-[#CCFF00]' : 'text-gray-500 hover:text-gray-300'}`}
           >
             <Wallet size={24} strokeWidth={activeTab === 'wallet' ? 2.5 : 2} />
             <span className="text-[9px] font-bold uppercase tracking-widest">Funds</span>
           </button>

           <button 
            onClick={() => setActiveTab('leaderboard')}
            className={`flex flex-col items-center space-y-1 transition-colors ${activeTab === 'leaderboard' ? 'text-[#CCFF00]' : 'text-gray-500 hover:text-gray-300'}`}
           >
             <Award size={24} strokeWidth={activeTab === 'leaderboard' ? 2.5 : 2} />
             <span className="text-[9px] font-bold uppercase tracking-widest">Rank</span>
           </button>

           <button 
            onClick={() => setActiveTab('profile')}
            className={`flex flex-col items-center space-y-1 transition-colors ${activeTab === 'profile' ? 'text-[#CCFF00]' : 'text-gray-500 hover:text-gray-300'}`}
           >
             <User size={24} strokeWidth={activeTab === 'profile' ? 2.5 : 2} />
             <span className="text-[9px] font-bold uppercase tracking-widest">ID</span>
           </button>
        </div>

      </div>
